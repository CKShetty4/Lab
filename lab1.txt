#include <stdio.h> 
#include <stdlib.h> 
#include <string.h> 
int main(int argc, char** argv) { 
 char input[101]; 
 int data[100]; 
 int i,count=0,parity,n; 
 
 printf("Enter the data:"); 
 scanf("%s",input); 
 n = strlen(input); 
 for (i=0;i<n;i++){ 
 if(input[i]!='0'&&input[i]!='1'){ 
 printf("Invalid input. Please enter only 0 or 1\n"); 
 return 1; 
 } 
data[i]=input[i]-'0'; 
 } 
 for (i=0;i<n;i++){ 
 if(data[i]==1){ 
 count++; 
 } 
 } 
 parity=(count%2==0)?0:1; 
 printf("Input data:"); 
 for (i=0;i<n;i++){ 
 printf("%d",data[i]); 
 } 
 printf("\n"); 
 printf("Calculated parity:%d\n",parity); 
 printf("Data with parity:"); 
 for (i=0;i<n;i++){ 
 printf("%d",data[i]); 
 } 
 printf("%d\n",parity); 
 return (EXIT_SUCCESS); 
}
